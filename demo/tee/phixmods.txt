function get_file_name(sequence path)
    for i=length(path) to 1 by -1 do
        if find(path[i],`\/`) then
            path = path[i+1..$]
            exit
        end if
    end for
    return path
end function

function get_file_base(sequence path)
    path = get_file_name(path)
    for i=length(path) to 1 by -1 do
        if path[i]='.' then
            path = path[1..i-1]
            exit
        end if
    end for
    return path
end function

function get_file_path(sequence path)
    for i=length(path) to 1 by -1 do
        if path[i]='\\' then
            path = path[1..i]
            exit
        end if
    end for
    return path
end function
